# 原型链

JavaScript 只有一种结构：对象。每个实例对象（ object ）都有一个私有属性（称之为 proto ）指向它的构造函数的原型对象（prototype ）。该原型对象也有一个自己的原型对象( proto ) ，层层向上直到一个对象的原型对象为 null。根据定义，null 没有原型，并作为这个原型链中的最后一个环节。

# 数组去重


deduplication (array) {
    if (!arr instanceof Array) {
      throw new Error ('Input need to be Array')
    }
    
    // alternative way 1 : SET
    let unique_array = Array.from(new Set(array))
    return unique_array

    // alternative way 2 : filter
    let unique_array = array.filter(function(elem, index, self) {
      return index === self.indexOf(elem)
    })
    return unique_array
    
    // alternative way 3: Object
    let obj = {}
    array.forEach((e) => {
        obj[e] = ''
    })
    return Object.keys(obj)
}

# ==和===

== 会做类型的隐式转换 ===不会

在比较基础类型的时候，== 会转化为统一类型然后比较值是否相等 === 会先比较类型，类型不一样就不相等

在比较高级类型（Array Object）的时候，两者没有区别，进行指针地址的比较 创建两个{a:111}时 === 为false

如果是基础类型和高级类型比较，== 会先把基础类型转换成高级类型 因为类型不同，=== 直接返回false

// 二层以上的对象依旧是复制用，非深拷贝

let deepClone = obj => {
	let newObj = Array.isArray(obj) ? []: {};
	if (obj && typeof obj === 'object') {
		for (let key in obj) {
			if (obj[key] && typeof obj[key] === 'object') {
				newObj[key] = deepClone(obj[key])
			} else {
				newObj[key] = obj[key]
			}
		}
	}
	return newObj
}

compose pipe 函数组合思想

let compose = function () {
	let args = [].slice.call(arguments)
	return function (x) {
		return args.reduceRight(function(res, cb) {
			return cb(res)
		}, x)
	}
}

# Javascript中防抖和节流

防抖： 停止触发事件之后的某一段时间后真正触发一次。 节流： 可持续触发，但固定事件内只触发一次

使用场景： 输入框，根据用户输入停顿进行请求

代码样例：

let deBounce = (fn, delay) => {
    let timer = null;
    return function (...args) {
        if (timer) {
            clearTimeout(timer);
        }
 
        timer = setTimeout(()=> {
            fn(...args);
        }, delay)
    }
}
 
throttle =  (fn, delay) => {
let flag = true
    let timer = null;
    return function (...args) {
        if (!flag) return
        flag = false
 
        setTimeout(()=> {
            fn(...args);
            flag = true
        }, delay)
    }
}
